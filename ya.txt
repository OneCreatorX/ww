local function gS(d,ts,nonce,round)
local env={
placeId=tostring(game.PlaceId),
jobId=game.JobId,
userId=tostring(lP.UserId),
version=version(),
round=tostring(round or 1),
timestamp=ts,
nonce=nonce
}
local baseStr=""
local keys={}
for k in pairs(d)do table.insert(keys,k)end
table.sort(keys)
for _,k in ipairs(keys)do
baseStr=baseStr..k.."="..tostring(d[k])..";"
end
local envKeys={}
for k in pairs(env)do table.insert(envKeys,k)end
table.sort(envKeys)
for _,k in ipairs(envKeys)do
baseStr=baseStr..k.."="..tostring(env[k])..";"
end
baseStr=baseStr..sS..";"..sK
local hash1=""
for i=1,#baseStr do
local ch=string.byte(baseStr,i)
hash1=hash1..string.char(((ch*17)%26)+97)
end
local hash2=""
for i=1,#hash1 do
local ch=string.byte(hash1,i)
hash2=hash2..string.char(((ch*23)%26)+97)
end
local finalSig=hash2:sub(1,16)..string.format("%02x",round or 1)..hash1:sub(-16)
return finalSig,env
end

local function sR(endpoint,data,round)
round=round or 1
local ts=tostring(fT())
local nonce=hS:GenerateGUID(false):gsub("-",""):sub(1,16)
data.placeId=tostring(game.PlaceId)
data.jobId=game.JobId
data.robloxVersion=version()
data.userId=tostring(lP.UserId)
data.round=round
data.timestamp=ts
local sig,env=gS(data,ts,nonce,round)
local requestData={
data=data,
nonce=nonce,
timestamp=ts,
round=round,
environment=env,
signature=sig
}
local headers={
["Content-Type"]="application/json",
["X-SecureAccess-Signature"]=sig,
["X-SecureAccess-Timestamp"]=ts,
["X-SecureAccess-Nonce"]=nonce,
["X-SecureAccess-Round"]=tostring(round),
["X-SecureAccess-Version"]="3.0",
["X-Roblox-PlaceId"]=tostring(game.PlaceId),
["X-Roblox-JobId"]=game.JobId,
["X-Roblox-UserId"]=tostring(lP.UserId)
}
local success,response=pcall(function()
return fR({
Url=wU..endpoint,
Method="POST",
Headers=headers,
Body=hS:JSONEncode(requestData)
})
end)
if not success then
return{success=false,message="Request failed",statusCode=0,round=round}
end
local decoded={}
pcall(function()decoded=hS:JSONDecode(response.Body)end)
return{
success=response.StatusCode==200 and decoded.success,
message=decoded.message or"Unknown error",
data=decoded.data,
statusCode=response.StatusCode,
challenge=decoded.challenge,
nextRound=decoded.nextRound,
token=decoded.token,
round=round
}
end

local function sC(challenge,round)
if not challenge then return nil end
local solution=nil
if challenge.type=="math"then
local expr=challenge.data
local func,err=loadstring("return "..expr)
if func then solution=func()end
elseif challenge.type=="hash"then
local input=challenge.data
local hash=""
local multiplier=19+(round or 1)*2
for i=1,#input do
local ch=string.byte(input,i)
hash=hash..string.char(((ch*multiplier)%26)+97)
end
solution=hash
elseif challenge.type=="roblox"then
if challenge.data=="placeId"then
solution=game.PlaceId
elseif challenge.data=="jobId"then
solution=game.JobId
elseif challenge.data=="userId"then
solution=lP.UserId
elseif challenge.data=="timestamp"then
solution=fT()
elseif challenge.data=="round"then
solution=round or 1
end
elseif challenge.type=="environment"then
if challenge.data=="camera"then
solution=workspace.CurrentCamera and"valid"or"invalid"
elseif challenge.data=="player"then
solution=lP and lP.Name or"unknown"
elseif challenge.data=="version"then
solution=version()
end
end
return solution
end

local vR=2

local function cA()
for round=1,vR do
local response=sR("/verify",{hwid=hw,playerID=tostring(lP.UserId),round=round},round)
if not response.success then
return false,nil,"Round "..round.." failed: "..response.message
end
if response.challenge then
local solution=sC(response.challenge,round)
if solution then
local challengeResponse=sR("/verify-challenge",{
hwid=hw,
playerID=tostring(lP.UserId),
challengeId=response.challenge.id,
solution=solution,
round=round
},round)
if not challengeResponse.success then
return false,nil,"Challenge round "..round.." failed"
end
response=challengeResponse
else
return false,nil,"Challenge solution failed for round "..round
end
end
if response.nextRound and round<vR then
task.wait(0.1)
continue
end
if round==vR then
local finalCheck=sR("/check",{hwid=hw,playerID=tostring(lP.UserId),round=round},round)
return finalCheck.success,finalCheck.data,finalCheck.message
end
end
return false,nil,"Multi-round verification incomplete"
end

local function qC()

return fR({
Url=wU.."/quick-check",
Method="POST",
Headers={["Content-Type"]="application/json"},
Body=hS:JSONEncode({hwid=hw,playerID=tostring(lP.UserId)})
})
end)
if success and response.StatusCode==200 then
local decoded={}
pcall(function()decoded=hS:JSONDecode(response.Body)end)
return decoded.success,decoded.data
end
return false,nil
end
